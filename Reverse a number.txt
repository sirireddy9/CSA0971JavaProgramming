import java.util.*;
class Reverse
{
	public static void main(String arg[]){
		int n,rem,rev;
		Scanner sc=new Scanner(System.in);
		System.out.println("enter n value:");
		n=sc.nextInt();
		rev=0;
		while(n!=0){
			rem=n%10;
			rev=(rev*10)+rem;
			n=n/10;
		}
		System.out.println("Reversed number:" +rev);
	}
}
