import java.util.*;
class Armstrong
{
	public static void main(String arg[]){
		int n,m,rem,sum=0;
		Scanner sc=new Scanner(System.in);
		System.out.println("enter n value:");
		n=sc.nextInt();
		m=n;
		while(n!=0){
			rem=n%10;
			sum+=(rem*rem*rem);
			n=n/10;
		}
		if(sum==m){
			System.out.println("Armstrong number");
		}
		else{
			System.out.println("not Armstrong number");
		}
	}
}
